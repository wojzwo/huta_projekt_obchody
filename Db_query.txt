use hutadb; 
create table Chip(
	chipId varchar(64) PRIMARY KEY,
	isEmployee bit(1)
);

create table Place(
	id int AUTO_INCREMENT PRIMARY KEY,
	name varchar(64),
	chipId varchar(64)
);

create table Mark(
	id int PRIMARY KEY,
	name varchar(64) CHARACTER SET utf8 COLLATE utf8_unicode_ci,
	isCommentRequired bit(1)
);

create table Employee(
	id int AUTO_INCREMENT PRIMARY KEY,
	name varchar(64),
	chipId varchar(64)
);

create table Track(
	id int AUTO_INCREMENT PRIMARY KEY,
	name varchar(64),
	creationDate date
);

create table TrackPlace(
	trackId int,
	placeId int,
	PRIMARY KEY(trackId, placeId)
);

create table Report(
	id int AUTO_INCREMENT PRIMARY KEY,
	trackId int,
	employeeId int,
	assignmentDate date,
	isFinished bit(1)
);

create table ReportPlace(
	reportId int,
	placeId int,
	status varchar(64),
	mark int,
	comment varchar(256),
	visitDate date
);
